(

    "GROUP_NM" character(50) COLLATE pg_catalog."default",
    "GROUP_FLG" character(2) COLLATE pg_catalog."default",
    "DEL_FLG" character(1) COLLATE pg_catalog."default",
    "ADD_DATE" date,
    "ADD_EMP_ID" character(10) COLLATE pg_catalog."default",
    "ADD_EMP_NAME" character(30) COLLATE pg_catalog."default",
    "UPDATE_DATE" date,
    "UPDATE_EMPT_ID" character(10) COLLATE pg_catalog."default",
    "UPDATE_EMP_NAME" character(30) COLLATE pg_catalog."default",
    "GROUP_ORDER" character(2) COLLATE pg_catalog."default",
    CONSTRAINT "M_GROUP_pkey" PRIMARY KEY ("GROUP_ID")
)
WITH (
    OIDS = FALSE
)
TABLESPACE pg_default;
